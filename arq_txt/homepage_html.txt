<script>
    let dadosCache = [];

    window.onload = () => carregarDados();

    async function carregarDados(idFocoManual = null) {
        try {
            const res = await fetch('/api/transacoes');
            dadosCache = await res.json();
            const corpo = document.getElementById('corpoTabela');
            corpo.innerHTML = '';

            let saldoAcumulado = 10000.00;

            dadosCache.forEach((item) => {
                const valorNumerico = parseFloat(item.valor) || 0;
                let colCredito = "---";
                let colDebito = "---";

                if (parseInt(item.tipo_cd) === 1) {
                    saldoAcumulado += valorNumerico;
                    colCredito = `<span class="txt-credito">${formatarMoeda(valorNumerico)}</span>`;
                } else {
                    saldoAcumulado -= valorNumerico;
                    colDebito = `<span class="txt-debito">${formatarMoeda(valorNumerico)}</span>`;
                }

                const tr = document.createElement('tr');
                tr.id = `linha-${item.id}`;
                tr.onclick = () => preencherFormulario(item.id);

                tr.innerHTML = `
                    <td>${item.id}</td>
                    <td>${formatarData(item.data)}</td>
                    <td>${item.favorecido || ''}</td>
                    <td class="alinhar-direita">${colCredito}</td>
                    <td class="alinhar-direita">${colDebito}</td>
                    <td class="col-saldo">${formatarMoeda(saldoAcumulado)}</td>
                `;
                corpo.appendChild(tr);
            });

            let idParaFocar = idFocoManual || (dadosCache.length > 0 ? dadosCache[dadosCache.length - 1].id : null);

            if (idParaFocar) {
                setTimeout(() => {
                    const linhaFoco = document.getElementById(`linha-${idParaFocar}`);
                    if (linhaFoco) {
                        linhaFoco.scrollIntoView({ behavior: 'smooth', block: 'center' });
                        linhaFoco.classList.add('flash-foco');
                        preencherFormulario(idParaFocar, false);
                    }
                }, 300);
            }
        } catch (e) { console.error("Erro MoneyBr:", e); }
    }

    function preencherFormulario(id, rolarParaForm = true) {
        const reg = dadosCache.find(d => d.id == id);
        if (!reg) return;

        document.querySelectorAll('tr').forEach(l => l.classList.remove('selecionada'));
        const linha = document.getElementById(`linha-${id}`);
        if (linha) linha.classList.add('selecionada');

        document.getElementById('tituloForm').innerText = "📝 Registro: " + reg.id;
        document.getElementById('tituloForm').style.backgroundColor = "#2980b9";

        document.getElementById('campoID').value = reg.id;
        document.getElementById('campoData').value = reg.data;
        document.getElementById('campoFavorecido').value = reg.favorecido || '';
        document.getElementById('campoTipo').value = reg.tipo_cd;
        document.getElementById('campoValor').value = formatarMoeda(reg.valor);
        document.getElementById('campoDescricao').value = reg.descricao || '';
        document.getElementById('campoCategoria').value = reg.categoria_id || 0;
        document.getElementById('campoSubCategoria').value = reg.subcategoria_id || 0;

        if (rolarParaForm) {
            document.getElementById('containerForm').scrollIntoView({ behavior: 'smooth' });
        }
    }

    function formatarMoeda(v) { return new Intl.NumberFormat('pt-BR', { minimumFractionDigits: 2, maximumFractionDigits: 2 }).format(v); }
    function formatarData(d) { if(!d) return ""; const [a, m, dia] = d.split('-'); return `${dia}/${m}/${a}`; }

    function mascaraMoeda(campo) {
        let v = campo.value.replace(/\D/g, '');
        v = (parseFloat(v) / 100);
        campo.value = isNaN(v) ? "" : formatarMoeda(v);
    }

    function limparForm() {
        document.getElementById('campoID').value = '';
        document.getElementById('containerForm').querySelectorAll('input').forEach(i => i.value = '');
        document.getElementById('campoTipo').value = '0';
        document.getElementById('tituloForm').innerText = "Novo Lançamento";
        document.getElementById('tituloForm').style.backgroundColor = "#34495e";
        document.querySelectorAll('tr').forEach(l => l.classList.remove('selecionada', 'flash-foco'));
    }

    async function confirmarSalvar() {
        const idExistente = document.getElementById('campoID').value;
        const valorRaw = document.getElementById('campoValor').value;
        const valorLimpo = parseFloat(valorRaw.replace(/\./g, '').replace(',', '.'));

        const payload = {
            id: idExistente,
            data: document.getElementById('campoData').value,
            favorecido: document.getElementById('campoFavorecido').value,
            tipo_cd: parseInt(document.getElementById('campoTipo').value),
            valor: valorLimpo,
            descricao: document.getElementById('campoDescricao').value,
            categoria_id: parseInt(document.getElementById('campoCategoria').value) || 0,
            subcategoria_id: parseInt(document.getElementById('campoSubCategoria').value) || 0
        };

        const rota = idExistente ? '/api/transacoes/atualizar' : '/api/transacoes/inserir';

        try {
            const res = await fetch(rota, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify(payload)
            });

            if (res.ok) {
                const resp = await res.json();
                const idFoco = idExistente || resp.id;
                await carregarDados(idFoco);
            }
        } catch (e) { alert("Erro de conexão."); }
    }
</script>
